# 4 重建二叉树

## 树的前序、中序、后序遍历

前序：根->左->右

中序：左->根->右

后序：左->右->根



## 解题思路

找到树的根结点以及根结点的左子树和右子树的前序、中序遍历vector递归建树，直到vector为空跳出递归。

```c++
/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> vin) {
        if(pre.empty() || vin.empty()){
            return nullptr;
        }
        TreeNode* head = new TreeNode(pre[0]);
        
        int root=0;
        int i;
        for(i=0;i<vin.size();i++){
            if(pre[0]==vin[i]){
                root = i;
                break;
            }
        }
        head->left = reConstructBinaryTree(vector<int>(pre.begin()+1,pre.begin()+1+root),
                                           vector<int>(vin.begin(),vin.begin()+root));
        head->right = reConstructBinaryTree(vector<int>(pre.begin()+root+1,pre.end()),
                                            vector<int>(vin.begin()+root+1,vin.end()));
        return head;
    }
};
```
